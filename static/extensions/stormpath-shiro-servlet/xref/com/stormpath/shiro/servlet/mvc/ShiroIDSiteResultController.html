11  
12  import javax.servlet.http.HttpServletRequest;
13  import javax.servlet.http.HttpServletResponse;
14  
15  
16  public class ShiroIDSiteResultController extends IdSiteResultController {
17  
18      @Override
19      protected ViewModel onAuthentication(HttpServletRequest request, HttpServletResponse response, AuthenticationResult result) {
20          ViewModel vm = super.onAuthentication(request, response, result);
21  
22          AuthenticationToken token = new StormpathWebRealm.AccessTokenResultAuthToken(result.getAccount());
23  
24          try {
25              SecurityUtils.getSubject().login(token);
26          } catch (AuthenticationException e) {
27              String msg = "Stormpath Shiro realm is not configured correctly, see documentation for: " + StormpathWebRealm.class.getName();
28              throw new ConfigurationException(msg, e);
29          }
30  
31          return vm;
32      }
33  }
