16  package com.stormpath.shiro.authz;
17  
18  import java.util.Set;
19  
20  /**
21   * A {@code PermissionsEditor} allows one to read and manipulate (append or remove) permissions on an underlying data
22   * structure.
23   * <p/>
24   * The primary default implementation of this interface is the {@link CustomDataPermissionsEditor}, which reads and
25   * modifies a Set of permissions stored in an {@link com.stormpath.sdk.directory.CustomData CustomData} instance.
26   *
27   * @see CustomDataPermissionsEditor
28   * @since 0.5
29   */
30  public interface PermissionsEditor {
31  
32      /**
33       * Adds a Shiro permission String to the associated Set of permission strings.
34       *
35       * @param perm the permission string to add to the associated Set of permission strings.
36       * @return this object for method chaining.
37       */
38      PermissionsEditor append(String perm);
39  
40      /**
41       * Removes the specified Shiro permission String from the associated Set of permission Strings.
42       *
43       * @param perm the permission string to remove from the associated Set of permission strings.
44       * @return this object for method chaining.
45       */
46      PermissionsEditor remove(String perm);
47  
48      /**
49       * Returns a read-only (immutable) view of the stored permission strings.  An immutable empty set will be returned
50       * if there are not any currently stored.
51       *
52       * @return an immutable view of the stored permission strings.
53       */
54      Set<String> getPermissionStrings();
55  }
