16  package com.stormpath.shiro.realm;
17  
18  import com.stormpath.sdk.directory.CustomData;
19  import com.stormpath.shiro.authz.CustomDataPermissionsEditor;
20  import org.apache.shiro.authz.Permission;
21  import org.apache.shiro.authz.permission.PermissionResolver;
22  import org.apache.shiro.authz.permission.WildcardPermissionResolver;
23  import org.apache.shiro.util.CollectionUtils;
24  
25  import java.util.Collections;
26  import java.util.HashSet;
27  import java.util.Set;
28  
29  /**
30   * A {@code CustomDataPermissionResolver} accesses a
31   * {@link #getCustomDataFieldName() specific named field} value of a {@link CustomData} resource that contains
32   * a {@code Set&lt;String&gt;} of Shiro permission Strings.
33   * <p/>
34   * The permissions stored in that field are assumed to be assigned to the CustomData's owning entity (an
35   * {@link com.stormpath.sdk.account.Account Account} or {@link com.stormpath.sdk.group.Group Group}).
36   * <h3>Custom Data Field Name</h3>
37   * You can configure what named field is used to store the permissions via
38   * {@link #setCustomDataFieldName(String) fieldName} property.
39   * <h3>String to Permission Conversion</h3>
40   * The Strings stored in the CustomData resource are converted to Shiro {@link Permission} instances via the
41   * {@link #getPermissionResolver() permissionResolver} property.  Unless overridden, the default instance is a Shiro
42   * {@link WildcardPermissionResolver}.
43   *
44   * @see AccountCustomDataPermissionResolver
45   * @see GroupCustomDataPermissionResolver
46   * @since 0.5
47   */
48  public class CustomDataPermissionResolver {
49  
50      private String customDataFieldName;
51  
52      private PermissionResolver permissionResolver;
53  
54      /**
55       * Creates a new instance, using the default {@link #getCustomDataFieldName() customDataFieldName} of
56       * {@code apacheShiroPermissions} and a default {@link WildcardPermissionResolver}.
57       */
58      public CustomDataPermissionResolver() {
59          this.customDataFieldName = CustomDataPermissionsEditor.DEFAULT_CUSTOM_DATA_FIELD_NAME;
60          this.permissionResolver = new WildcardPermissionResolver();
61      }
62  
63      /**
64       * Returns the name of the {@link CustomData} field used to store the {@code Set&lt;String&gt;}
65       * of permissions.  The default name is
66       * {@code apacheShiroPermissions}, implying a {@code CustomData} JSON representation as follows:
67       * <pre>
68       * {
69       *     ... any other of your own custom data properties ...,
70       *
71       *     "apacheShiroPermissions": [
72       *         "perm1",
73       *         "perm2",
74       *         ...,
75       *         "permN"j
76       *     ]
77       * }
78       * </pre>
79       * You can change the name by calling {@link #setCustomDataFieldName(String)}.
80       *
81       * @return the name of the {@link CustomData} field used to store the {@code Set&lt;String&gt;}
82       *         of permissions.
83       */
84      public String getCustomDataFieldName() {
85          return customDataFieldName;
86      }
87  
88      /**
89       * Sets the name of the {@link CustomData} field used to store the {@code Set&lt;String&gt;}
90       * of permissions.  The default name is
91       * {@code apacheShiroPermissions}, implying a {@code CustomData} JSON representation as follows:
92       * <pre>
93       * {
94       *     ... any other of your own custom data properties ...,
95       *
96       *     "apacheShiroPermissions": [
97       *         "perm1",
98       *         "perm2",
99       *         ...,
100      *         "permN"j
101      *     ]
102      * }
103      * </pre>
104      * If you changed this name to be {@code myApplicationPermissions} for example, the CustomData representation
105      * would look something like this instead:
106      * <pre>
107      * {
108      *     ... any other of your own custom data properties ...,
109      *
110      *     "myApplicationPermissions": [
111      *         "perm1",
112      *         "perm2",
113      *         ...,
114      *         "permN"j
115      *     ]
116      * }
117      * </pre>
118      *
119      * @param customDataFieldName the name of the {@link CustomData} field used to store the {@code Set&lt;String&gt;}
120      *                            of permissions.
121      */
122     public void setCustomDataFieldName(String customDataFieldName) {
123         this.customDataFieldName = customDataFieldName;
124     }
125 
126     /**
127      * Returns the {@link PermissionResolver} used to convert {@link #getCustomDataFieldName() stored permission strings}
128      * to Shiro {@link Permission} instances for {@link org.apache.shiro.subject.Subject Subject} authorization.
129      * <p/>
130      * The default / pre-configured instance is a {@link WildcardPermissionResolver}.
131      *
132      * @return the {@link PermissionResolver} used to convert {@link #getCustomDataFieldName() stored permission strings}
133      *         to Shiro {@link Permission} instances for {@link org.apache.shiro.subject.Subject Subject} authorization.
134      */
135     public PermissionResolver getPermissionResolver() {
136         return permissionResolver;
137     }
138 
139     /**
140      * Sets the {@link PermissionResolver} used to convert {@link #getCustomDataFieldName() stored permission strings}
141      * to Shiro {@link Permission} instances for {@link org.apache.shiro.subject.Subject Subject} authorization.
142      *
143      * @param permissionResolver the {@link PermissionResolver} used to convert {@link #getCustomDataFieldName() stored permission strings}
144      *                           to Shiro {@link Permission} instances for {@link org.apache.shiro.subject.Subject Subject} authorization.
145      */
146     public void setPermissionResolver(PermissionResolver permissionResolver) {
147         this.permissionResolver = permissionResolver;
148     }
149 
150     /**
151      * Returns a {@code Set&lt;String&gt;} of permission strings that are stored in the specified
152      * {@link CustomData} instance (under the {@link #getCustomDataFieldName() customDataFieldName} key), or an
153      * empty collection if no permissions are stored.
154      * <p/>
155      * This implementation internally delegates field access and Set construction to a
156      * {@link CustomDataPermissionsEditor} instance, e.g.
157      * <pre>
158      * return new CustomDataPermissionsEditor(customData)
159      *     .setFieldName(getCustomDataFieldName())
160      *     .getPermissionStrings();
161      * </pre>
162      *
163      * @param customData the custom data instance that might have a {@code Set&lt;String&gt;} of permissions stored in
164      *                   a key named {@link #getCustomDataFieldName()}.
165      * @return a {@code Set&lt;String&gt;} of permission strings that are stored in the specified
166      *         {@link CustomData} instance (under the {@link #getCustomDataFieldName() customDataFieldName} key), or an
167      *         empty collection if no permissions are stored.
168      * @see CustomDataPermissionsEditor
169      */
170     protected Set<String> getPermissionStrings(CustomData customData) {
171         return new CustomDataPermissionsEditor(customData)
172                 .setFieldName(getCustomDataFieldName())
173                 .getPermissionStrings();
174     }
175 
176     /**
177      * Returns a set of Shiro {@link Permission} instances stored in the specified {@link CustomData} resource.  This
178      * implementation will:
179      * <ol>
180      * <li>{@link #getPermissionStrings(com.stormpath.sdk.directory.CustomData) Get all permission strings} stored
181      * in the CustomData instance</li>
182      * <li>Loop over these strings, and for each one, create a {@link Permission} instance using the
183      * {@link #getPermissionResolver() permissionResolver} property.</li>
184      * <li>Return the total constructed Set of Permission instances to the caller.</li>
185      * </ol>
186      *
187      * @param customData the CustomData instance that may contain permission strings to obtain
188      * @return a set of Shiro {@link Permission} instances stored in the specified {@link CustomData} resource.
189      */
190     protected Set<Permission> getPermissions(CustomData customData) {
191 
192         Set<String> permStrings = getPermissionStrings(customData);
193 
194         if (CollectionUtils.isEmpty(permStrings)) {
195             return Collections.emptySet();
196         }
197 
198         PermissionResolver permissionResolver = getPermissionResolver();
199 
200         Set<Permission> perms = new HashSet<Permission>(permStrings.size());
201 
202         for (String s : permStrings) {
203             Permission perm = permissionResolver.resolvePermission(s);
204             perms.add(perm);
205         }
206 
207         return perms;
208     }
209 }
