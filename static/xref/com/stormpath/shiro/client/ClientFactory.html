30  /**
31   * A simple bridge component that allows a Stormpath SDK Client to be created via Shiro's
32   * {@link org.apache.shiro.util.Factory Factory} concept.
33   * <p/>
34   * As this class is a simple bridge between APIs, it does not do much - all configuration properties are immediately
35   * passed through to an internal {@link com.stormpath.sdk.client.ClientBuilder ClientBuilder} instance, and the
36   * {@link #createInstance()} implementation merely calls {@link com.stormpath.sdk.client.ClientBuilder#build()}.
37   * <h5>Usage</h5>
38   * Example {@code shiro.ini} configuration:
39   * <p/>
40   * <pre>
41   * [main]
42   * ...
43   * cacheManager = some.impl.of.org.apache.shiro.cache.CacheManager
44   * securityManager.cacheManager = $cacheManager
45   *
46   * stormpathClient = com.stormpath.shiro.client.ClientFactory
47   * stormpathClient.apiKeyFileLocation = /home/myhomedir/.stormpath/apiKey.properties
48   * stormpathClient.cacheManager = $cacheManager
49   *
50   * stormpathRealm = com.stormpath.shiro.realm.ApplicationRealm
51   * stormpathRealm.client = $stormpathClient
52   * stormpathRealm.applicationRestUrl = https://api.stormpath.com/v1/applications/yourAppIdHere
53   *
54   * securityManager.realm = $stormpathRealm
55   *
56   * ...
57   * </pre>
58   *
59   * @see ClientBuilder
60   * @see ClientBuilder#setApiKey(com.stormpath.sdk.api.ApiKey)
61   * @see com.stormpath.sdk.api.ApiKeyBuilder#setFileLocation(String)
62   * @since 0.1
63   */
64  public class ClientFactory extends AbstractFactory<Client> {
65  
66      private ClientBuilder clientBuilder;
67  
68      public ClientFactory() {
69          super();
70          this.clientBuilder = Clients.builder();
71      }
72  
73      @Override
74      protected Client createInstance() {
75          return clientBuilder.build();
76      }
77  
78      public ClientBuilder getClientBuilder() {
79          return clientBuilder;
80      }
81  
82      public void setClientBuilder(ClientBuilder clientBuilder) {
83          this.clientBuilder = clientBuilder;
84      }
85  
86      /**
87       * Calls {@code clientBuilder.setApiKey(ApiKeys.builder().}{@link com.stormpath.sdk.api.ApiKeyBuilder#setFileLocation(String) setFileLocation(apiKeyFileLocation)}{@code .build())}.
88       * See that JavaDoc for expected syntax/format.
89       *
90       * @param apiKeyFileLocation the file, classpath or url location of the API Key {@code .properties} file to load when
91       *                           constructing the API Key to use for communicating with the Stormpath REST API.
92       * @see ClientBuilder#setApiKey(com.stormpath.sdk.api.ApiKey)
93       * @see com.stormpath.sdk.api.ApiKeyBuilder#setFileLocation(String)
94       * @since 0.2
95       */
96      public void setApiKeyFileLocation(String apiKeyFileLocation) {
97          this.clientBuilder.setApiKey(ApiKeys.builder().setFileLocation(apiKeyFileLocation).build());
98      }
99  
100     /**
101      * Calls {@code clientBuilder.setApiKey(ApiKeys.builder().}{@link com.stormpath.sdk.api.ApiKeyBuilder#setInputStream(java.io.InputStream) setInputStream(apiKeyInputStream)}{@code .build())}.
102      *
103      * @param apiKeyInputStream the InputStream to use to construct a configuration Properties instance.
104      * @see ClientBuilder#setApiKey(com.stormpath.sdk.api.ApiKey)
105      * @see com.stormpath.sdk.api.ApiKeyBuilder#setInputStream(java.io.InputStream)
106      * @since 0.2
107      */
108     public void setApiKeyInputStream(InputStream apiKeyInputStream) {
109         this.clientBuilder.setApiKey(ApiKeys.builder().setInputStream(apiKeyInputStream).build());
110     }
111 
112     /**
113      * Calls {@code clientBuilder.setApiKey(ApiKeys.builder().}{@link com.stormpath.sdk.api.ApiKeyBuilder#setReader(java.io.Reader) setReader(apiKeyReader)}{@code .build())}.
114      *
115      * @param apiKeyReader the reader to use to construct a configuration Properties instance.
116      * @see ClientBuilder#setApiKey(com.stormpath.sdk.api.ApiKey)
117      * @see com.stormpath.sdk.api.ApiKeyBuilder#setReader(java.io.Reader)
118      * @since 0.2
119      */
120     public void setApiKeyReader(Reader apiKeyReader) {
121         this.clientBuilder.setApiKey(ApiKeys.builder().setReader(apiKeyReader).build());
122     }
123 
124     /**
125      * Calls {@code clientBuilder.setApiKey(ApiKeys.builder().}{@link com.stormpath.sdk.api.ApiKeyBuilder#setProperties(java.util.Properties) setProperties(properties)}{@code .build())}.
126      *
127      * @param properties the properties instance to use to load the API Key ID and Secret.
128      * @see ClientBuilder#setApiKey(com.stormpath.sdk.api.ApiKey)
129      * @see com.stormpath.sdk.api.ApiKeyBuilder#setProperties(java.util.Properties)
130      * @since 0.2
131      */
132     public void setApiKeyProperties(Properties properties) {
133         this.clientBuilder.setApiKey(ApiKeys.builder().setProperties(properties).build());

137      * Calls {@code clientBuilder.setApiKey(ApiKeys.builder().}{@link com.stormpath.sdk.api.ApiKeyBuilder#setIdPropertyName(String) setIdPropertyName(apiKeyIdPropertyName)}{@code .build())}.
138      *
139      * @param apiKeyIdPropertyName the name used to query for the API Key ID from a Properties instance.

148     /**
149      * Calls {@code clientBuilder.setApiKey(ApiKeys.builder().}{@link com.stormpath.sdk.api.ApiKeyBuilder#setSecretPropertyName(String) setSecretPropertyName(apiKeySecretPropertyName)}{@code .build())}.
150      *
151      * @param apiKeySecretPropertyName the name used to query for the API Key Secret from a Properties instance.
152      * @see ClientBuilder#setApiKey(com.stormpath.sdk.api.ApiKey)
153      * @see com.stormpath.sdk.api.ApiKeyBuilder#setSecretPropertyName(String)

156     public void setApiKeySecretPropertyName(String apiKeySecretPropertyName) {

158     }
159 
160     /**
161      * Uses the specified Shiro {@link CacheManager} instance as the Stormpath SDK Client's CacheManager, allowing both
162      * Shiro and Stormpath SDK to share the same cache mechanism.
163      * <p/>
164      * If for some reason you don't want to share the same cache mechanism, you can explicitly set a Stormpath SDK-only
165      * {@link com.stormpath.sdk.cache.CacheManager CacheManager} instance via the
166      * {@link #setStormpathCacheManager(com.stormpath.sdk.cache.CacheManager) setStormpathCacheManager} method.
167      *
168      * @param cacheManager the Shiro CacheManager to use for the Stormpath SDK Client's caching needs.
169      * @since 0.4.0
170      * @see #setStormpathCacheManager(com.stormpath.sdk.cache.CacheManager)
171      */
172     public void setCacheManager(CacheManager cacheManager) {
173         com.stormpath.sdk.cache.CacheManager stormpathCacheManager =
174                 new ShiroCacheManager(cacheManager);
175         this.clientBuilder.setCacheManager(stormpathCacheManager);
176     }
177 
178     /**
179      * Calls {@code clientBuilder.}{@link ClientBuilder#setCacheManager(com.stormpath.sdk.cache.CacheManager) setCacheManager}
180      * using the specified Stormpath {@link com.stormpath.sdk.cache.CacheManager CacheManager} instance, but <b>note:</b>
181      * This method should only be used if the Stormpath SDK should use a <em>different</em> CacheManager than what
182      * Shiro uses.
183      * <p/>
184      * If you prefer that Shiro and the Stormpath SDK use the same cache mechanism to reduce complexity/configuration,
185      * configure your preferred Shiro {@code cacheManager} first and then use that cacheManager by calling the
186      * {@link #setCacheManager(org.apache.shiro.cache.CacheManager) setCacheManager(shiroCacheManager)} method
187      * instead of this one.
188      *
189      * @param cacheManager the Storpmath SDK-specific CacheManager to use for the Stormpath SDK Client's caching needs.
190      * @since 0.4.0
191      * @see #setCacheManager(org.apache.shiro.cache.CacheManager)
192      */
193     public void setStormpathCacheManager(com.stormpath.sdk.cache.CacheManager cacheManager) {
194         this.clientBuilder.setCacheManager(cacheManager);
195     }
196 }
