1   package com.stormpath.shiro.servlet.config;
2   
3   import com.stormpath.sdk.impl.config.PropertiesSource;
4   import com.stormpath.shiro.sdk.servlet.config.impl.MultipleSourceConfigFactory;
5   
6   import javax.servlet.ServletContext;
7   import java.util.ArrayList;
8   import java.util.Collection;
9   
10  /**
11   * Subclass of {@link MultipleSourceConfigFactory} that adds shiro.ini properties as config properties. <BR/>
12   * <BR/>
13   * The Shiro {@link org.apache.shiro.config.Ini} must be available as a {@link ServletContext} attribute named
14   * {@link StormpathShiroConfigFactory#SHIRO_STORMPATH_PROPERTIES_ATTRIBUTE}.
15   */
16  public class StormpathShiroConfigFactory extends MultipleSourceConfigFactory {
17  
18      final public static String SHIRO_STORMPATH_PROPERTIES_ATTRIBUTE = StormpathShiroConfigFactory.class.getName() + "_SHIRO_STORMPATH_PROPERTIES";
19  
20      @Override
21      protected Collection<PropertiesSource> getInternalPropertiesSources(ServletContext servletContext) {
22  
23          PropertiesSource shiroPropertySource = (PropertiesSource) servletContext.getAttribute(SHIRO_STORMPATH_PROPERTIES_ATTRIBUTE);
24  
25          Collection<PropertiesSource> sources = new ArrayList<PropertiesSource>();
26  
27          if (shiroPropertySource!= null) {
28              sources.add(shiroPropertySource);
29          }
30  
31          return sources;
32      }
33  }
